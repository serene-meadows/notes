-- default/ zero value
--
  Each type has a zero-value/ default value
  constants have to be initialized on declearation,thus no concept of zero-initialized


  -- common zero values
  -- 
    integer                - 0
    floating point numbers - 0
    complex numbers        - 0 + 0i
    strings                - ""


    arrays  - elements are zero-initialized
    structs - fields are zero-initialized


    maps            - nil
    slices          - nil
    channels        - nil
    pointers        - nil
    functions       - nil
    error interface - nil


    json - when unmarshalling into a struct
           its fields are zero initialized
