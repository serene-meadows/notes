-- defined types/ named type
--
  note: 
      aliases are not a new type, they are just another name for the same type

  ```go
    package main

    import "fmt"

    /*
      CURRENT GO SPEC: NAMED TYPES
      A "Named Type" is any type identified by a name (an identifier). 
      If you can refer to it with a single word, it is likely a Named Type.
      
      Categories:
      1. Predeclared: int, float64, string, bool, error, etc.
      2. Defined: Created by you (e.g., 'type MyInt int').
      3. Instantiated: A generic type given specific arguments (e.g., 'List[int]').
      4. Type Parameters: The 'T' placeholder inside a generic function/struct.
    */

    type Age int          // Named Type (Defined)
    type User struct {     // Named Type (Defined)
        Name string
    }

    /*
      CURRENT GO SPEC: UNNAMED TYPES
      A type that has no name and is instead defined by its structure (type literal).
      All unnamed types are COMPOSITE types.
      
      Common Examples:
      - Pointers: *int, *User
      - Slices: []string, []int
      - Maps: map[string]int
      - Channels: chan bool
      - Arrays: [5]int
      - Inline Structs: struct{ ID int }
    */

    func main() {
        // --- 1. Strictness Between Named Types ---
        // Rule: Two different named types ALWAYS require explicit conversion, 
        // even if their underlying structure is identical.
        
        var count int = 10 
        var myAge Age = 25 
        
        // count = myAge   // ERROR: Cannot use Age as int
        count = int(myAge) // SUCCESS: Must convert explicitly
        _ = count

        // --- 2. Flexibility With Unnamed Types ---
        // Rule: An Unnamed type can be assigned to a Named type (and vice versa) 
        // without conversion, as long as the underlying structure matches.

        // 'data' is an UNNAMED type (literal struct)
        var data = struct {
            Name string
        }{Name: "Alice"}

        // 'u' is a NAMED type (User)
        var u User

        u = data // SUCCESS: Underlying structures match, and 'data' is unnamed.
        _ = u

        // --- 3. Composite Types are Unnamed ---
        var p *int    // Unnamed (Pointer)
        var s []int   // Unnamed (Slice)
        
        fmt.Println("Notes: Named = Identifier | Unnamed = Literal Structure")
    }
  ```
