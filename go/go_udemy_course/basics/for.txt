syntax
  for initialization; condition; post {
    // code block
  }
  all 3 
    initialization, condition, post are optional

  if condition is absent, it is treated as true
    for initialization; ; post {} <-> for initialization; true; post {}




c-style loop
  for i := 0; i < 10; i++ {
    // code block
  }

while loop
  for i < 10 {
    // code block
  }

infinite loop
  for {
    // code block
  }




for i := 1; i < 6; i++ {
  fmt.Println(i)

  i := i // i now shadows the outer variable
  i := 1
  _ = i
}
  // output 
  // 1
  // 2
  // 3
  // 4 
  // 5


range loop
  slices/ arrays -> returns index, value
  maps -> returns key, value
  integers -> range 10 -> returns 0 .. 9

iterating over a collection
  numbers := []int{1, 2, 3, 4, 5}
  for index, value := range numbers {
    fmt.Printf("Index: %d, Value: %d", index, value)
  }


range over integers
  for i := range 10 {
    // code block
  }
    // range -> [0, 10)
