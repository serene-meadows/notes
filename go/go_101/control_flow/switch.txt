The `switch-case` block is a multi-way conditional execution structure.
It provides a cleaner and more readable alternative to complex `if-else` chains.





-- The Full Syntax and Anatomy
--
  switch InitSimpleStatement; CompareOperand {
  case ExpressionList1:
      // execute if CompareOperand equals one of the expressions in ExpressionList1
  case ExpressionList2:
      // ...
  default:
      // execute if no cases match
  }
    

  -- Rules for the Operands
  --
    1. CompareOperand: Evaluated to a typed value (or untyped converted to default).
       Because it needs a type, it CANNOT be the untyped `nil` identifier.
    2. ExpressionList: A comma-separated list of expressions. Each expression
       must be comparable to the `CompareOperand`.


  -- Optional Components (Omissions)
  --
    Both the Initialization Statement and the Compare Operand are optional.

    ```go
      func omissionExamples() {
        // 1. Full form
        switch n := 5; n {
        case 5: // ...
        }



        // 2. Init statement omitted (semicolon is dropped)
        switch 5 {
        case 5: // ...
        }



        // 3. Compare operand omitted (Defaults to `true`)
        switch _ = 5; {
        case true: // ...
        }



        // 4. Both omitted (Behaves like a cleaner `if-else` chain)
        // This is highly idiomatic in Go.
        x := 10
        switch { // Equivalent to: `switch true {`
        case x > 5:
          // matches because x > 5 is true
        case x < 5:
          // ...
        }
      }

    ```





-- The `default` Branch
--
  - You can have at most ONE `default` branch.
  - Unlike some other languages, the `default` branch can appear ANYWHERE in
    the switch block (top, middle, or bottom). Its position does not affect
    the execution logic.

  example
    ```go
      func defaultPositioning() {
        n := rand.Intn(3)

        // All three of these switch blocks are logically identical:

        switch n {
        case 0:  fmt.Println("Zero")
        case 1:  fmt.Println("One")
        default: fmt.Println("Two") // Standard position
        }


        switch n {
        default: fmt.Println("Two") 
        case 0:  fmt.Println("Zero")
        case 1:  fmt.Println("One")
        }


        switch n {
        case 0:  fmt.Println("Zero")
        default: fmt.Println("Two")  
        case 1:  fmt.Println("One")
        }

      }

    ```






-- Implicit Break & Duplicate Cases
--
  -- Implicit Break
  --
    In C, C++, and Java, execution "falls through" to the next case automatically
    unless you explicitly write `break`.
    In Go, execution AUTOMATICALLY breaks out of the switch block at the end of
    a matched case branch. You do not need (and generally shouldn't use) `break`.


  -- Duplicate Cases
  --
    The Go compiler will throw an error if it detects duplicate constant values
    across different cases.


  example
    ```go
      func implicitBreakExample() {
        switch n := rand.Intn(100); n % 9 {
        case 0:
          fmt.Println("Multiple of 9")
          // Automatically breaks here. No `break` statement needed.


        case 1, 2, 3: // Comma-separated list acts as an "OR"
          fmt.Println("Mod 9 is 1, 2, or 3")
          break // This is allowed, but completely unnecessary (nonsense).


        case 4, 5, 6:
          fmt.Println("Mod 9 is 4, 5, or 6")


        // case 6, 7, 8:
          // COMPILE ERROR: duplicate case 6 in switch
          // Go protects you from overlapping constant cases.


        default:
          fmt.Println("Mod 9 is 7 or 8")
        }
      }

    ```





-- The `fallthrough` Keyword
--
  Because Go breaks automatically, if you explicitly WANT execution to slip into
  the very next case block (bypassing the next case's condition check), you must
  use the `fallthrough` keyword.


  -- Strict Rules for `fallthrough`
  --
    1. It MUST be the absolute final statement in a branch.
    2. It CANNOT be used in the final branch of the entire switch block


  example
    ```go
      func fallthroughSuccess() {
        switch n := rand.Intn(100) % 5; n {
        case 0, 1, 2, 3, 4:
          fmt.Println("n =", n)

          
          // Forces execution to jump into the `case 5, 6, 7, 8:` block below,
          // regardless of whether `n` actually matches 5, 6, 7, or 8.
          fallthrough 
          

        case 5, 6, 7, 8:
          // We can declare a new shadowed variable here.
          n := 99 
          fmt.Println("shadowed n =", n) // Prints 99
          

          // Slips into the default block.
          fallthrough 

          
        default:
          // This uses the original `n` from the switch initialization.
          fmt.Println("original n =", n) 
        }
      }




      func fallthroughErrors() {
        switch n := rand.Intn(5); n {
        case 1:
          fmt.Println("n = 1")
          // ERROR: The `if` block is the final statement, not `fallthrough`.
          /*
          if true {
            fallthrough // COMPILE ERROR: fallthrough statement out of place
          }
          */

          
        case 2:
          // ERROR: `_ = n` is the final statement, not `fallthrough`.
          //
          // fallthrough // COMPILE ERROR: fallthrough statement out of place
          // _ = n

          
        default:
          fmt.Println(n)
          // ERROR: Cannot use fallthrough in the very last branch of a switch.
          // 
          // fallthrough // COMPILE ERROR: cannot fallthrough final case in switch
        }
      }

    ```
