-- if - with initialization
--
  go allows you to execute a short initialization statment before evaluating the boolean condition 
    its useful, because it keeps the variable local to the scope 

    if assignment; condition {
      // codeblock
    }

    ```go
      if val := getVal(); val > 10 {
          fmt.Println(val, "is large")
      } else {
          fmt.Println(val, "is small")
      }

      // fmt.Println(val) // ERROR: val is undefined here!

    ```





-- shadow if condition variable
--
  ```go
    if h < 0 {
      h := h // the left variable h now shadows the outer variable h
      _ = h
    }

  ```
