absolute path
  complete path from the root directory

relative path
  path relative from current working directory

  uses
    . to go down the current working directory
    .. to go up the current working directory


go provides
  path/filepath package
    which handles platform specific path issues
    and has utilties for path manipulation

usefull functions
  filepath.Join
    // constucts a path by joining individual components with appropriate seperator

  filepath.Split
    // splits a path into directory and file name components

  filepath.Clean
    // normalizes a path, cleaning up redundant seperators, and resolving . and .. references

  filepath.Abs
    // returns absolute path of a relative path

  filepath.IsAbs
    // bool, if absulate path

  filepath.Base
    // retreives that last element of the path, which is the file, or maybe a directory

  filepath.Dir
    // retreives the directory part of the path

  filepath.Ext 
    // retreives extension of file

  filepath.Rel
    // gives a path that is relative from the perspective of a target to the perspective of a source


  os.Mkdir()
  os.MkdirAll()
    // creating directories 

  os.Remove()
  os.RemoveAll()
    // deleting directories

example 
  ````go
    relativePath := "./data/file.txt"
    absolutePath := "/home/user/docs/file.txt"

    // join paths using filepath.Join
    joinedPath := filepath.Join("home", downloads", "file.zip") // "home/downloads/file.zip"
      // takes variadic argument 

    normalizedPath := filepath.Clean("./data/..data/file.txt") // "data/file.txt"

    dir, file := filepath.Split("/home/user/docs/file.txt")
      // dir -> "/home/user/docs"
      // file -> "file.txt"

    base := filepath.Base("/home/user/docs/file.txt") // file.txt

    fmt.Println(filepath.IsAbs(relativePath)) // false
    fmt.Println(filepath.IsAbs(absulutePath)) // true

    fmt.Println(filepath.Ext(file))


    rel, err = filepath.Rel("a/b", "a/b/t/file")
      // basepath -> a/b
      // target -> a/b/t/file
    if err != nil {
      panic(err)
    }
    fmt.Println(rel) // t/file


    rel, err = filepath.Rel("a/c", "a/b/t/file")
      // basepath -> a/c
      // target -> a/b/t/file
    if err != nil {
      panic(err)
    }
    fmt.Println(rel) // ../b/t/file
  ````
