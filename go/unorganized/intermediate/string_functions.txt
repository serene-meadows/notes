  carefull note 
    ascii -> require 1 byte 
    for other characters (read abouit runes) , go uses utf-8 encoding so they might take more than 1 byte (upto 4)
    the reason this is important is because, some functions work with bytes (not chars directly)

// regex 
  in the "regexp" package
  explained with an example 
    ````go
      // base string 
        strName := "Hello, 123 45 Go!"
      
      // regex expression 
      re := regex.MustCompile(`\d+`)  // one or more digits
        // returns a regex
        // MustCompiler -> panicks if error
        // regex is enclosed in ` (back ticks) // since we need raw string literals

      matches := re.FindAllString(strName) 
          // int -> max number of matches we want
            // value of -1 -> all matches
        // returns a slice of matches 
          // nil if no matches

      fmt.Println(matches) // [123 45]
    ````

// string builder 
  in the "strings" package
  // strings are immutable
  // string builder is designed to concatenate strings more efficiently
    // normal string concatenation requires building intermediate strings which can be memory and processing intensive

  string builder can be used directly after declaration without initialization 
  a builder can be reused by calling the reset method, which clears it's internal buffer 
  the final string can be retreived using a string method 

  declaration and usage 
    ````go
      var builder strings.Builder 

      // write 
        // .Write()
        // .WriteString()
        // .WriteRune()
        // .WriteByte()
      builder.WriteString("hello ")
      builder.WriteString("world ")
      builder.WriteString("what's ")
      builder.WriteString("up ")

      // convert to a string 
      result := builder.String() // "hello world what's up "

      // using WriteRune() to add a char 
      builder.WriteRune('-')
      result = builder.String() // "hello world what's up -"

      // builder keeps on writing to its existing string buffer 
      // so in order to start a new string, you have to reset 
      builder.Reset()
      builder.WriteString("Starting fresh")

      result = builder.String() // "Starting fresh"
